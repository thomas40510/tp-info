import numpy as np
import matplotlib.pyplot as plt

def xn(n, t):
    return( 4 * np.cos(t) - np.cos(n * t) )
    
def yn(n, t):
    return( 4 * np.sin(t) - np.sin(n * t) )

def Trace_Cn(N):
    T = np.linspace(0, 2 * np.pi, 200)
    couleurs = ['grey', 'red', 'black', 'green', 'orange', 'blue']
    plt.figure('Traces_Cn')
    plt.title('Traces des courbes Cn pour n allant de ' + str(N[0]) + ' à ' + str(N[-1]))
    for n in N:
        Xn = [xn(n, t) for t in T]
        Yn = [yn(n, t) for t in T]
        plt.plot(Xn, Yn, color = couleurs[n % len(couleurs)])
    plt.grid()
    plt.show()

import scipy.integrate as integr

def V(n, t):
    return( np.array([[-4 * np.sin(t) + n * np.sin(n * t)], [4 * np.cos(t) - n * np.cos(n * t)]]) )

def long(n):
    return( integr.quad(lambda t : alg.norm(V(n, t)), 0, 2 * np.pi) )

def Visu_long(N):
    Ln = []
    for n in N:
        Ln.append(long(n))
