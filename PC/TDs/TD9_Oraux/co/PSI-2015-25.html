def LPn(n):
    '''n >= 1, renvoie la liste des polynomes [P1, ..., Pn]
    la complexite est lineaire'''
    LP = [Polynomial([1]), Polynomial([0, 1])]
    if n == 1:
        return(LP)
    LP.append( Polynomial([-1, 0, 1]) )

        return(LP)

    return(LP)
