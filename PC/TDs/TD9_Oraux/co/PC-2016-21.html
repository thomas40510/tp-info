def An(a, b, n):
    An = np.eye(n)
    for c in range(1, n):
        An[0, c] = (c % 2) * b + ((c + 1) % 2) * a
    for l in range(1, n):
        An[l, 0] = (l % 2) * b + ((l + 1) % 2) * a
    return(An)
