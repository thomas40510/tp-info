## Importation des modules

import numpy as np
import matplotlib.pyplot as plt


## Déclaration des constantes

v0_kmh = 130        # vitesse initiale en km/h
v0 = v0_kmh / 3.6   # vitesse initiale en m/s
mu = 0.9            # facteur de frottement roues/sol
M = 3e3             # masse du fourgon en kg
E = 3               # empattement du véhicule en m
h = 0.7             # hauteur du centre de gravité du véhicule G en m
d = 1.7             # distance entre l'essieu avant et G en m
H = 3               # hauteur du véhicule en m
L = 1.8             # largeur du véhciule en m
rho = 1.2           # masse volumique de l'air en kg/m^3
Cx = 0.4            # coefficient de trainée pour le fourgon
g = 9.81            # accélération de la pesanteur en m/s^2

## Déclaration des fonctions

def Surf_ref(H, L):
    '''surface de référence ou maître couple de la forme, approchée par la 
    formule de Paul Frère.'''
    return( 0.85 * H * L )

def trainee(rho, S, Cx, v):
    '''trainée de la forme.'''
    return( 0.5 * rho * S * Cx * pow(v, 2) )

def eqdf(v, mu, g, M, rho, S, Cx):
    return( - mu * g - (1 / M) * trainee(rho, S, Cx, v) )

def EQDF(X, T, mu = 0.9, g = 9.81, M = 3e3, rho = 1.2, H = 3, L = 1.8, Cx = 0.4):
    v = X[1]
    S = Surf_ref(H, L)
    return( np.array([v, eqdf(v, mu, g, M, rho, S, Cx)]) )

## Initialisation

X0 = np.array([0, v0])

# estimation du temps de freinage dans le cas d'un freinage sans frottement
tf = v0 / (mu * g)
# choix de la durée de l'intervalle de temps
tmax = int(1.1 * tf) + 1

n = 200     # nombre d'instants considérés
T = np.linspace(0, tmax, n)

## Estimation de la solution

def Euler_exp(X0, T, Equadiff):
    XX = [X0]
    nt = len(T)
    for i in range(1, nt):
        X_estim = XX[i - 1] + \
        (T[i] - T[i - 1]) * Equadiff(XX[i - 1], T[i - 1])
        if X_estim[1] >= 0:
            XX.append(X_estim)
        else:
            XX.append([XX[i - 1][0], 0])
    return(XX)

## Tracés des évolutions

XX = Euler_exp(X0, T, EQDF)

Distances = [elt[0] for elt in XX]
Vitesses = [elt[1] for elt in XX]

plt.figure('Distance')
plt.plot(T, Distances, color = 'red')
plt.xlabel('temps en secondes')
plt.ylabel('distance de freinage en m')
plt.title('Evolution de la distance de freinage')
plt.grid()
plt.savefig('Evo_dist_freinage.png')
plt.show()

plt.figure('Vitesse')
